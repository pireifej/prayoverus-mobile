# Push OTA update to PREVIEW build only (your test device, safe for testing)
cd expo-app && npx eas update --branch preview --message "test"

# Push OTA update to ALL users (production - only when ready to release)
cd expo-app && npx eas update --branch production --message "description of changes"

# Build preview APK for Android (one-time per device, ~15 min)
# Install the .apk on your test device, then use eas update --branch preview after that
cd expo-app && npx eas build --profile preview --platform android

# Build for Android (production - for Google Play Store)
cd expo-app && npx eas build --platform android

# Build for iOS (production - requires Apple Developer account $99/year)
cd expo-app && npx eas build --platform ios

# Submit to Google Play Store
cd expo-app && npx eas submit --platform android

# Submit to Apple App Store
cd expo-app && npx eas submit --platform ios
